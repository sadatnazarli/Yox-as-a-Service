import requests
import json

base_url = "{{ request.base_url if request else 'http://127.0.0.1:5000' }}" # Replace with your API base URL

def get_yox_behane(category="yox", ad=None):
    """
    Fetches an excuse from the YOX API.

    Args:
        category (str): The excuse category (default: 'yox').
        ad (str, optional): The user's name. Defaults to None.

    Returns:
        str: The excuse text or an error message.
    """
    url = f"{base_url}/{category}"
    params = {}
    if ad:
        params['ad'] = ad

    try:
        response = requests.get(url, params=params)
        response.raise_for_status() # Raise an exception for HTTP errors

        data = response.json()
        return data.get("cavab", "Invalid response format.")

    except requests.exceptions.RequestException as e:
        return f"Request error: {e}"
    except json.JSONDecodeError:
        return "Response is not in JSON format."
    except Exception as e:
        return f"Unexpected error: {e}"

# Example usage

async function getYoxBehane(category = 'yox', ad = null) {
    const baseUrl = "{{ request.base_url if request else 'http://127.0.0.1:5000' }}"; // Replace with your API base URL
    let url = `${baseUrl}/${category}`;

    if (ad) {
        const params = new URLSearchParams({ ad: ad });
        url += `?${params.toString()}`;
    }

    try {
        const response = await fetch(url);

        if (!response.ok) {
            const errorData = await response.json();
            throw new Error(errorData.xeta || `HTTP error! Status: ${response.status}`);
        }

        const data = await response.json();
        return data.cavab || 'Invalid response format.';

    } catch (error) {
        console.error('Fetch error:', error);
        return `Error occurred: ${error.message}`;
    }
}
